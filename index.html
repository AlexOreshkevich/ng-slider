Single value

Code

<!DOCTYPE html>
  <html>
  <head>
    ...
    <link href="css/ng-slider.min.css" rel="stylesheet" />
    <script type="javascripts/javascript" src="angular.min.js" />
    <script type="javascripts/javascript" src="ng-slider.min.js" />
  </head>
  <body>
    ...
    <div>
      <input ng-model="value"  type="text" id="mySlider1" slider options="options" />
    </div>
    
    <script>
      function ctrl($scope) {
        $scope.value = "50";
        $scope.options = {       
          from: 1,
          to: 100,
          step: 1,
          dimension: " km"        
        };
      }   
    </script>
        
  </body>
  </html>
